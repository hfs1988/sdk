package main

import (
	"net/http"

	"sdk/adapters/rest"
)

type Router interface {
	Get(queryString string, handler http.HandlerFunc)
	ListenAndServe()
}

func main() {
	var router Router = rest.GetRouterInstance(3000)
	router.Get("/", func(w http.ResponseWriter, r *http.Request) {
		w.Write([]byte("welcome"))
	})
	router.ListenAndServe()
}
